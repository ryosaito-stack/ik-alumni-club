  ) {}

      this.converter(doc.id, doc.data())
    );

  async getById(id: string): Promise<T | null> {
    const docRef = doc(db, this.collectionName, id);
    const docSnap = await getDoc(docRef);
    if (docSnap.exists()) {
      return this.converter(docSnap.id, docSnap.data());
    }
    return null;
  }
}
